Construa um algoritmo que leia dois valores numéricos inteiros e efetue a adição; caso o resultado seja maior que 10, apresentá-lo 

#include "stdio.h"

int main(){

    int n1;
    int n2;

    printf("Digite n1:\n");
    scanf("%d", &n1);
    printf("Digite n2:\n");
    scanf("%d", &n2);

    int soma = n1+n2;

    if(soma > 10){
        printf("Soma: %d", soma);
    }
}
